83 void precompute()
84 {
85 
86  if (!useProjectedCalculus)
87  calculus = new PolyCalculus(surfmesh);
88  else
89  {
90  //Using the projection embedder
91  auto params2 = SH3::defaultParameters() | SHG3::defaultParameters() | SHG3::parametersGeometryEstimation();
92  params2("r-radius", (double) radiusII);
93  auto surfels = SH3::getSurfelRange( surface, params2 );
94  iinormals = SHG3::getIINormalVectors(binary_image, surfels,params2);
95  trace.info()<<iinormals.size()<<std::endl;
96  auto myProjEmbedder = [&](Face f, Vertex v)
97  {
98  const auto nn = iinormals[f];
99  RealPoint centroid(0.0,0.0,0.0); //centroid of the original face
100  auto cpt=0;
101  for(auto v: surfmesh.incidentVertices(f))
102  {
103  cpt++;
104  centroid += surfmesh.position(v);
105  }
106  centroid = centroid / (double)cpt;
107  RealPoint p = surfmesh.position(v);
108  auto cp = p-centroid;
109  RealPoint q = p - nn.dot(cp)*nn;
110  return q;
111  };
112  psMesh->addFaceVectorQuantity("II normals", iinormals);
113  calculus = new PolyCalculus(surfmesh);
114  calculus->setEmbedder( myProjEmbedder );
115  }
116 
117  heat = new GeodesicsInHeat<PolyCalculus>(calculus);
118 
119  if (!skipReg)
120  {
121  calculusReg = new PolyCalculus(surfmeshReg);
122  heatReg = new GeodesicsInHeat<PolyCalculus>(calculusReg);
123  }
124  trace.beginBlock("Init solvers");
125  heat->init(dt);
126  if (!skipReg)
127  heatReg->init(dt);
128  trace.endBlock();
129 }
