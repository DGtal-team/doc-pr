71 void precompute()
72 {
73  auto projEmbedder = [&]( int f, int v)
74  {
75  auto nn = surfmesh.faceNormal( f );
76  RealPoint centroid = surfmesh.faceCentroid( f );
77  RealPoint p = surfmesh.position( v );
78  const auto cp = p - centroid;
79  RealPoint q = p - nn.dot(cp)*nn;
